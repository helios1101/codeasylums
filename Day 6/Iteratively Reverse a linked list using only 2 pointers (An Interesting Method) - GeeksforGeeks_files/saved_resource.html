              this can also work using only two pointers

              static void reverse(struct Node** head_ref)
              {

              struct Node* current = *head_ref;
              struct Node* next;
              while (current->next != NULL)
              {
              next = current->next;
              current->next = next->next;
              next->next=(*head_ref);
              *head_ref=next;
              }

              }

                                          There is another approach to do this using two pointer :

                                          void reverseUsingTwoPointer(struct node **head){
                                          if(!(*head))
                                          return;
                                          struct node *circularListLast,*next;
                                          circularListLast=*head;
                                          next=circularListLast->next;
                                          circularListLast->next=*head;
                                          while(next){
                                          circularListLast->next=next;
                                          next=next->next;
                                          circularListLast->next->next=*head;
                                          *head=circularListLast->next;
                                          }
                                          circularListLast->next=NULL;
                                          }
